Exercício 1 -

#include <stdio.h>



int main()

{

  int numA, numB, numX, numY; // Variaveis para os calculos propostos



  printf("Vamos realizar duas contas simples, uma de adição e outra de subtração. \n");

  printf("Digite o primeiro número: ");

  scanf("%i", &numA);

  printf("Digite o segundo número: ");

  scanf("%i", &numB);

  printf("O resultado da sua conta de adicão será: \n");

  printf("%i + %i = %i", numA, numB, numA + numB);

  printf("\n");

  printf("Agora, faremos uma conta de subtração, ok? \n");

  printf("Para isso, informe outros dois números distintos! Qual será o primeiro? ");

  scanf("%i", &numX);

  printf("E qual será o segundo número? ");

  scanf("%i", &numY);

  printf("O resultado da subtração desses números será: \n");

  printf("%i - %i = %i", numX, numY, numX - numY);

  return 0;

}



Exercício 2:

#include <stdio.h>



int main()

{

  float numA, numB, media; // Variavies que serão utilizadas

   

  printf("Vamos agora tirar a média entre dois números. \n");

  printf("Digite o primeiro número: ");

  scanf("%f", &numA);

  printf("Digite o segundo número: ");

  scanf("%f", &numB);

  media = (numA + numB) / 2; // CALCULO DA MEDIA ENTRE OS NUMEROS

  printf("A média entre %.1f e %.1f é %.1f", numA, numB, media);



  return 0;

}



Exercício 3:

#include <stdio.h>



int main()

{

  int numX; // Variavel usada na conta

   

  printf("Agora, iremos descobrir os números sucessores e antecessores de um número informado por você, vamos lá?");

  printf("\n");

  printf("Para isso, preciso que você me informe o número que iremos verificar: ");

  scanf ("%i", &numX);

  printf("\n");

  printf("O sucessor de %i é %i", numX, numX + 1);

  printf("\n");

  printf("E o antecessor de %i é %i", numX, numX - 1);

  printf("\n");

  printf("Ou seja, a ordem é assim: %i, %i, %i", numX - 1, numX, numX + 1);



  return 0;

}
